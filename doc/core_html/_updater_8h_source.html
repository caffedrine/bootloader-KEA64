98 struct updater_operation_t
99 {
100  updater_enum_t operation_desc;
101  updater_task_vector_t tasks;
102  uint32_t current_task;
103  bool user_stopped;
104 
105  uint32_t current()
106  {
107  uint32_t current = 0;
108 
109  for (uint32_t i=0; i < current_task; ++i)
110  {
111  current += tasks[i].total;
112  }
113 
114  current += tasks[current_task].current;
115 
116  return current;
117  }
118 
119  uint32_t total()
120  {
121  uint32_t total = 0;
122  for (uint32_t i=0; i < tasks.size(); ++i)
123  {
124  total += tasks[i].total;
125  }
126  return total;
127  }
128 
129  updater_operation_t(updater_enum_t operation_desc)
130  : operation_desc(operation_desc), tasks(), current_task(0), user_stopped(false) {}
131 
132  updater_operation_t(uint32_t value, const char * description)
133  : operation_desc(value, description), tasks(), current_task(0), user_stopped(false) {}
134 };
