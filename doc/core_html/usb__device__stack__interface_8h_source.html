121 #define for_each_ep_in_class(epPtr, usbclassPtr, classtype) \
122  for(uint32_t class_i = 0; usbclassPtr[class_i].type != USB_CLASS_INVALID; class_i++) \
123  if((usbclassPtr[class_i].type == classtype) || (classtype == USB_CLASS_ALL)) \
124  for(uint32_t intf_i = 0; intf_i < usbclassPtr[class_i].interfaces.count; intf_i++) \
125  for(uint32_t ep_i = 0; \
126  ((ep_i < usbclassPtr[class_i].interfaces.interface[intf_i].endpoints.count) && \
127  ((epPtr = usbclassPtr[class_i].interfaces.interface[intf_i].endpoints.ep + ep_i) != NULL)); \
128  ep_i++)
