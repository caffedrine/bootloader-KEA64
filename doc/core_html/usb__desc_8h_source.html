249  uint8_t bEndpointAddress; /* Endpoint # 0 - 15 | IN/OUT */

251  #define OUT_ENDPOINT (0x00) /* OUT endpoint, host to device */
252  #define ENDPOINT_MASK (0x0F) /* Mask endpoint # */

254  #define CONTROL_ENDPOINT (0x00) /* Control transfers */
255  #define ISOCH_ENDPOINT (0x01) /* Isochronous transfers */
256  #define BULK_ENDPOINT (0x02) /* Bulk transfers */

261  #define ISOCH_ASYNC (0x04) /* Asynchronous */
262  #define ISOCH_ADAPT (0x08) /* Adaptive */

264  #define ISOCH_DATA (0x00) /* Data endpoint */
265  #define ISOCH_FEEDBACK (0x10) /* Feedback endpoint */
266  #define ISOCH_IMPLICIT (0x20) /* Implicit feedback */
267  #define ISOCH_RESERVED (0x30) /* Reserved */
268  uint8_t wMaxPacketSize[2]; /* Bits 10:0 = max. packet size */
269  /* For high-speed interrupt or isochronous only, additional

271  #define PACKET_SIZE_MASK (0x7FF) /* packet size bits */
272  #define NO_ADDITONAL (0x0000) /* 1 / microframe */
